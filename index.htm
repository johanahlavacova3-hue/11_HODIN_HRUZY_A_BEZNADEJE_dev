  <style>

    body {

      text-align: center;

      margin-top: 50px;

    }

    .bar {

      width: 300px;

      height: 20px;

      background: #555;

      margin: 10px auto;

      border-radius: 10px;

      overflow: hidden;

    }

    .fill {

      height: 100%;

      width: 100%;

      background: limegreen;

      transition: width 0.2s;

    }

  </style>

<body>

  <h1>Kořenový vězeň Zdena</h1>

  

  <div id="game">

    <p>Zdraví Zdeny:</p>

    <div class="bar"><div id="zdenaHealth" class="fill"></div></div>

    

    <p>Kořen HP:</p>

    <div class="bar"><div id="rootHealth" class="fill"></div></div>

    

    <p>Mačkej mezerník, abys rozbil kořen. SHIFT = křičení.</p>

  </div>

  

  <script>

    let zdenaHP = 100;

    let rootHP = 100;

    let gameOver = false;


    const zdenaBar = document.getElementById("zdenaHealth");

    const rootBar = document.getElementById("rootHealth");


    // Zdenovi ubývá zdraví postupně

    setInterval(() => {

      if (!gameOver) {

        zdenaHP -= 1;

        if (zdenaHP < 0) zdenaHP = 0;

        zdenaBar.style.width = zdenaHP + "%";

        if (zdenaHP <= 0) endGame("Prohra: Kořen tě pohltil!");

      }

    }, 1000);


    // Ovládání kláves

    document.addEventListener("keydown", (e) => {

      if (gameOver) return;


      if (e.code === "Space") {

        rootHP -= 5;

        if (rootHP < 0) rootHP = 0;

        rootBar.style.width = rootHP + "%";

        if (rootHP <= 0) endGame("Vyhrál jsi: Rozbil jsi kořen!");

      }


      if (e.code === "ShiftLeft") {

        if (Math.random() < 0.1) { // 10% šance, že přijdou kamarádi

          endGame("Vyhrál jsi: Kamarádi tě zachránili!");

        }

      }

    });


    function endGame(message) {

      gameOver = true;

      alert(message);

    }

  </script>

</body>

</html>
